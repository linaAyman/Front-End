   random(myarray) {
    let ctr = myarray.length, temp, index;
    while (ctr > 0) {
        index = Math.floor(Math.random() * ctr);
        ctr--;
        temp = myarray[ctr];
        myarray[ctr] = myarray[index];
        myarray[index] = temp;
    }
    return myarray;
}
